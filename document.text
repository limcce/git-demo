vim <file>创建文件并编辑
git status 查看状态
git add <file>添加文件到暂存区
git commit -m "日志信息" <file>将文件提交到本地仓库
git rm --cached <file>将文件从暂存区删除
git log 查看详细的版本信息
git reflog 查看简化的版本信息
cat <file>查看文件内容
git reset --hard <版本号> 穿梭版本
ll 查看当前目录下的文件

git branch <分支名> 创建新的分支
git branch -v查看所有分支
git switch/checkout <分支名> 穿梭到指定分支(switch和checkout都行)
git merge <分支名> 合并某分支(此时不能在要合并的分支中)
git remote -v查看别名
git remote 简易版查看别名
git remote add <别名> http地址
